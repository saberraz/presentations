\subsection{Proprietary software}
\begin{frame}{Proprietary software}
	\begin{block}{Pros and Cons}
		\begin{itemize}
			\item Advantages:
			\begin{itemize}
				\item Unified platform
				\item Support and documentation
				\item Usability
			\end{itemize}
			\item Disadvantages
			\begin{itemize}
				\item Dependency
				\item Overall cost
				\item Flexibility
			\end{itemize}
		\end{itemize}
	\end{block}
\end{frame}

\subsection{Black box OS software}
\begin{frame}{Black box OS software}
	\begin{block}{Pros and Cons}
		\begin{itemize}
			\item Advantages:
			\begin{itemize}
				\item Things work
				\item Easy to migrate away
			\end{itemize}
			\item Disadvantages
			\begin{itemize}
				\item Similar cost to Proprietary Software
				\item Not configurable/Customisable
			\end{itemize}
		\end{itemize}
	\end{block}
\end{frame}
