			\item 1 month of bug fix

\subsection{LTR, LR, DEV}
\begin{frame}{QGIS release - LTR, LR, DEV}
	\begin{block}{Give choice}
		\begin{itemize}
			\item LTR (LTS): If you want reliability
			\item LR: If you want new features and don't your house on fire!
			\item DEV: Current development branch
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}{How to support QGIS (LTR)?}

		\begin{itemize}
			\item Participate
			\begin{itemize}
				\item Testing
				\begin{itemize}
					\item Run multiple QGIS versions
					\item Test previous bugs
					\item Try new features
					\item Help with other reported bugs
				\end{itemize}
				\item Filing bugs (https://hub.qgis.org/projects/quantum-gis/issues):
				\begin{itemize}
					\item Step-by-step guide to reproduce your bug
					\item Follow-ups
					\item Provide test data
				\end{itemize}
			\end{itemize}
		\end{itemize}
